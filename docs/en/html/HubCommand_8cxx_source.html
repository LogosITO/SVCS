51bool HubCommand::execute(const std::vector<std::string>& args) {
52 if (args.empty()) {
53 notifyError("Repository path is required");
54 return false;
55 }
56
57 if (args.size() > 1) {
58 notifyError("Too many arguments");
59 return false;
60 }
61
62 fs::path repo_path(args[0]);
63 if (repo_path.empty()) {
64 notifyError("Invalid repository path");
65 return false;
66 }
67
68 if (repo_path.is_relative()) {
69 repo_path = fs::absolute(repo_path);
70 }
71
72 fs::path repo_dir = repo_path / ".svcs";
73
74 if (fs::exists(repo_dir)) {
75 notifyError("Repository already exists at: " + repo_dir.string());
76 return false;
77 }
78
79 if (!fs::exists(repo_path)) {
80 fs::create_directories(repo_path);
81 }
82
83 notifyInfo("Creating hub repository at " + repo_dir.string());
84
85 try {
86 fs::create_directories(repo_dir / "objects");
87 fs::create_directories(repo_dir / "refs" / "heads");
88 fs::create_directories(repo_dir / "refs" / "tags");
89 fs::create_directories(repo_dir / "hooks");
90 fs::create_directories(repo_dir / "info");
91
92 std::ofstream config(repo_dir / "config");
93 config << "[core]\n"
94 << "bare = true\n"
95 << "repositoryformatversion = 0\n"
96 << "[svcs]\n"
97 << "hub = true\n";
98 config.close();
99
100 std::ofstream head(repo_dir / "HEAD");
101 head << "ref: refs/heads/main\n";
102 head.close();
103
104 std::ofstream description(repo_dir / "description");
105 description << "Unnamed repository; edit this file to name the repository.\n";
106 description.close();
107
108 } catch (const std::exception& e) {
109 notifyError(std::string("Initialization failed: ") + e.what());
110 fs::remove_all(repo_dir);
111 return false;
112 }
113
114 notifySuccess("Hub repository successfully created at " + repo_dir.string());
115 return true;
116}
