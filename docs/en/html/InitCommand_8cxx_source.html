18bool InitCommand::execute(const std::vector<std::string>& args) {
19 std::string path = ".";
20 bool force = false;
21 bool showHelpFlag = false;
22
23 for (const auto& arg : args) {
24 if (arg == "--force" || arg == "-f") {
25 force = true;
26 } else if (arg == "--help" || arg == "-h") {
27 showHelpFlag = true;
28 } else if (arg.find("-") == 0) {
29 // Неизвестная опция
30 eventBus_->notify({Event::WARNING_MESSAGE,
31 "Unknown option: " + arg, "init"});
32 } else {
33 // Это путь (может быть ".")
34 path = arg;
35 }
36 }
37
38 if (showHelpFlag) {
39 showHelp();
40 return true;
41 }
42
43 eventBus_->notify({Event::GENERAL_INFO,
44 "Initializing new repository in: " + path, "init"});
45
46 if (force) {
47 eventBus_->notify({Event::GENERAL_INFO,
48 "Force flag enabled - will reinitialize if repository exists", "init"});
49 }
50
51 if (repoManager_->initializeRepository(path, force)) {
52 eventBus_->notify({Event::REPOSITORY_INIT_SUCCESS,
53 "Initialized empty SVCS repository in " + path + "/.svcs", "init"});
54 return true;
55 } else {
56 eventBus_->notify({Event::ERROR_MESSAGE,
57 "Failed to initialize repository", "init"});
58 return false;
59 }
60}
