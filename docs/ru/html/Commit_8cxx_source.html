43std::string Commit::serialize() const {
44 std::stringstream ss;
45 ss << "tree " << tree_hash << "\n";
46
47 for(const auto& parent_hash : parent_hashes) {
48 ss << "parent " << parent_hash << "\n";
49 }
50
51 ss << "author " << author << " " << timestamp << " +0000\n";
52 ss << "committer " << author << " " << timestamp << " +0000\n";
53
54 ss << "\n";
55 ss << message;
56
57 return ss.str();

60Commit Commit::deserialize(const std::string& raw_content) {
61 std::istringstream ss(raw_content);
62 std::string line;
63
64 std::string thash;
65 std::vector<std::string> phashes;
66 std::string author_line;
67 std::time_t tstamp = 0; // Будет извлечен из author_line/committer_line
68
69 // --- 1. Чтение Заголовка ---
70 while (std::getline(ss, line) && !line.empty()) {
71
72 size_t space_pos = line.find(' ');
73 if (space_pos == std::string::npos) {
74 continue;
75 }
76
77 std::string key = line.substr(0, space_pos);
78 std::string value = line.substr(space_pos + 1);
79
80 if (key == "tree") {
81 thash = value;
82 } else if (key == "parent") {
83 phashes.push_back(value);
84 } else if (key == "author") {
85 author_line = std::move(value);
86 } else if (key == "committer") {
87 std::string committer_line;
88 committer_line = std::move(value);
89 }
90 }
91
92 std::string author_data; // "Имя <email>"
93
94 // Функция для парсинга поля, чтобы избежать дублирования кода
95 auto parse_user_field = [&](const std::string& field_value, std::string& out_name, std::time_t& out_time) {
96 if (field_value.empty()) return;
97
98 // Поиск последнего пробела (перед часовым поясом)
99 size_t tz_start_pos = field_value.rfind(' ');
100 if (tz_start_pos == std::string::npos) return;
101
102 // Поиск пробела перед меткой времени (timestamp)
103 size_t ts_start_pos = field_value.rfind(' ', tz_start_pos - 1);
104 if (ts_start_pos == std::string::npos) return;
105
106 // 1. Извлекаем метку времени
107 std::string ts_str = field_value.substr(ts_start_pos + 1, tz_start_pos - ts_start_pos - 1);
108 try {
109 out_time = static_cast<std::time_t>(std::stoll(ts_str));
110 } catch (const std::exception& e) {
111 throw std::runtime_error("Commit deserialization error: Failed to parse timestamp: " + std::string(e.what()));
112 }
113
114 // 2. Извлекаем имя и email
115 out_name = field_value.substr(0, ts_start_pos);
116 // Удаляем завершающий пробел между email и timestamp (если есть)
117 if (!out_name.empty() && out_name.back() == ' ') {
118 out_name.pop_back();
119 }
120 };
121
122 // Парсим author_line. Здесь out_name - это author_data, out_time - это tstamp.
123 parse_user_field(author_line, author_data, tstamp);
124
125 // --- 3. Чтение Сообщения ---
126 std::string message;
127 if (ss.peek() != std::char_traits<char>::eof()) {
128 // Читаем все оставшееся содержимое до конца файла (EOF)
129 message.assign(
130 (std::istreambuf_iterator<char>(ss)),
131 std::istreambuf_iterator<char>()
132 );
133 }
134
135 // Удаляем потенциально добавленный пустой строкой перевод строки в начале сообщения
136 if (!message.empty() && message.front() == '\n') {
137 message.erase(0, 1);
138 }
139
140 // ВАЖНО: Мы должны вернуть объект, используя только те данные,
141 // которые фактически хранятся в Commit.
142
143 if (thash.empty() || author_data.empty()) {
144 throw std::runtime_error("Commit deserialization error: Missing mandatory field (tree_hash or author).");
145 }
146
147 return {
148 std::move(thash),
149 std::move(phashes),
150 std::move(author_data), // Передаем только имя/email
151 std::move(message),
152 tstamp
153 };
154}
