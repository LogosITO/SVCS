232 void detach(std::shared_ptr<IObserver> observer_to_remove) override {
233 std::lock_guard<std::mutex> lock(observers_mutex_);
234
235 auto it = std::remove_if(observers_.begin(), observers_.end(),
236 [&](const std::weak_ptr<IObserver>& weak_obs) {

238 return shared_obs && (shared_obs == observer_to_remove);
239 }
240 );
241 observers_.erase(it, observers_.end());

257 void notify(const Event& event) const override {
258 std::string color = getEventColor(event.type);
259 std::string icon = getEventIcon(event.type);
260 std::string sourceColor = ConsoleColor::DIM + ConsoleColor::BLACK;
261
262 // Format message with colors
263 std::string formattedMessage = color + icon + " " + event.details +
264 sourceColor + " [" + event.source_name + "]" +
265 ConsoleColor::RESET;
266
267 // Create event copy with formatted message
268 Event coloredEvent = event;
269 coloredEvent.details = formattedMessage;
270
271 for (auto& observer : observers_) {
272 observer.lock()->update(coloredEvent);
273 }
274 }
