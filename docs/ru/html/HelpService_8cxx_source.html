41void HelpService::showCommandHelp(const std::string& commandName) const {

50bool HelpService::commandExists(const std::string& commandName) const {
51 auto commands = getAvailableCommands();
52 return std::find(commands.begin(), commands.end(), commandName) != commands.end();
53}

55std::string HelpService::getCommandUsage(const std::string& commandName) const {
56 if (getUsageCallback_) {
57 return getUsageCallback_(commandName);
58 }
59
60 // Fallback: try to get usage from description if available
61 std::string description = getCommandDescription(commandName);
62 if (description != "Unknown command" && description != "Help service not properly initialized") {
63 return "svcs " + commandName + " [arguments]";
64 }
65
66 return "";
67}
