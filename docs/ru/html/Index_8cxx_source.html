26void Index::getFileMetaData(const fs::path& full_path, long long& size, fs::file_time_type& mtime) {
27 try {
28 if (fs::exists(full_path) && !fs::is_directory(full_path)) {
29 size = fs::file_size(full_path);
30 mtime = fs::last_write_time(full_path);
31 } else {
32 size = 0;
33 mtime = fs::file_time_type::min();
34 }
35 } catch (const fs::filesystem_error& e) {
36 std::cerr << "Filesystem error getting metadata for " << full_path.string() << ": " << e.what() << std::endl;
37 size = 0;
38 mtime = fs::file_time_type::min();
39 }
40}

64void Index::stage_file(const fs::path& relative_path) {
65 fs::path full_path = repo_root_path / relative_path;
66
67 if (!fs::exists(full_path) || fs::is_directory(full_path)) {
68 throw std::runtime_error("Cannot stage: path is invalid or is a directory: " + full_path.string());
69 }
70
71 std::string file_content;
72 try {
73 file_content = read_file_to_string(full_path);
74 } catch (const std::exception& e) {
75 throw std::runtime_error("Error reading file content: " + std::string(e.what()));
76 }
77
78 Blob blob(file_content);
79 std::string blob_hash = blob.getHashId();
80
81 storage_.saveObject(blob);
82
83 IndexEntry new_entry;
84 long long current_size;
85 fs::file_time_type current_mtime;
86 getFileMetaData(full_path, current_size, current_mtime);
87
88 new_entry.file_path = relative_path;
89 new_entry.blob_hash = blob_hash;
90 new_entry.file_size = current_size;
91 new_entry.last_modified = current_mtime;
92
93 addEntry(new_entry);
94 save();
95}

97bool Index::isFileModified(const fs::path& file_path) const {
98 const IndexEntry* entry = getEntry(file_path);
99 fs::path full_path = repo_root_path / file_path;
100
101 // Файл удален
102 if (entry && !fs::exists(full_path)) {
103 return true;
104 }
105
106 // Новый файл
107 if (!entry && fs::exists(full_path) && !fs::is_directory(full_path)) {
108 return true;
109 }
110
111 // Файл существует и есть в индексе
112 if (entry && fs::exists(full_path) && !fs::is_directory(full_path)) {
113 long long current_size;
114 fs::file_time_type current_mtime;
115 getFileMetaData(full_path, current_size, current_mtime);
116
117 // Проверяем размер
118 if (current_size != entry->file_size) {
119 return true;
120 }
121
122 // Проверяем время модификации
123 using std::chrono::time_point_cast;
124 using std::chrono::seconds;
125
126 bool time_differs = current_mtime != entry->last_modified;
127
128 if (time_differs) {
129 // Если время изменилось, проверяем хеш
130 try {
131 std::string current_content = read_file_to_string(full_path);
132 Blob current_blob(current_content);
133 std::string current_hash = current_blob.getHashId();
134
135 if (current_hash != entry->blob_hash) {
136 return true;
137 }
138 } catch (const std::exception& e) {
139 return true;
140 }
141 }
142
143 return false;
144 }
145
146 return false;
147}

149void Index::save() const {
150 std::ofstream ofs(index_file_path, std::ios::out | std::ios::trunc);
151 if (!ofs.is_open()) {
152 throw std::runtime_error("Error: Could not open index file for writing.");
153 }
154
155 for (const auto& pair : entries) {
156 const IndexEntry& entry = pair.second;
157
158 using std::chrono::time_point_cast;
159 using std::chrono::seconds;
160
161 auto duration = time_point_cast<seconds>(entry.last_modified).time_since_epoch();
162 long long mtime_ticks = duration.count();
163
164 ofs << entry.blob_hash << " "
165 << entry.file_size << " "
166 << mtime_ticks << " "
167 << entry.file_path.generic_string() << "\n";
168 }
169}

171void Index::load() {
172 if (!fs::exists(index_file_path)) {
173 return;
174 }
175
176 std::ifstream ifs(index_file_path, std::ios::in);
177 if (!ifs.is_open()) {
178 throw std::runtime_error("Error: Could not open index file for reading.");
179 }
180
181 entries.clear();
182
183 std::string line;
184 while (std::getline(ifs, line)) {
185 std::stringstream ss(line);
186 IndexEntry entry;
187
188 if (!(ss >> entry.blob_hash >> entry.file_size)) {
189 continue;
190 }
191
192 long long mtime_ticks;
193 if (!(ss >> mtime_ticks)) {
194 continue;
195 }
196
197 std::string path_part;
198 std::getline(ss, path_part);
199
200 if (!path_part.empty() && path_part[0] == ' ') {
201 path_part = path_part.substr(1);
202 }
203
204 if (path_part.empty()) {
205 continue;
206 }
207
208 entry.file_path = path_part;
209
210 using std::chrono::seconds;
211 entry.last_modified = fs::file_time_type(seconds(mtime_ticks));
212
213 entries[entry.file_path] = entry;
214 }
215}

217bool Index::write() const {
218 std::ofstream ofs(index_file_path, std::ios::out | std::ios::trunc);
219 if (!ofs.is_open()) {
220 throw std::runtime_error("std::runtime_error If the index file cannot be written.");
221 return false;
222 }
223
224 for (const auto& pair : entries) {
225 const IndexEntry& entry = pair.second;
226
227 using std::chrono::time_point_cast;
228 using std::chrono::seconds;
229
230 auto duration = time_point_cast<seconds>(entry.last_modified).time_since_epoch();
231 long long mtime_ticks = duration.count();
232
233 ofs << entry.blob_hash << " "
234 << entry.file_size << " "
235 << mtime_ticks << " "
236 << entry.file_path.generic_string() << "\n";
237 }
238
239 if (ofs.fail()) {
240 return false;
241 }
242
243 return true;
244}

246std::string Index::createTreeObject() {
247 if (entries.empty()) {
248 throw std::runtime_error("Cannot create a commit: the index is empty.");
249 }
250
251 std::map<fs::path, std::vector<IndexEntry>> staged_by_directory;
252
253 for (const auto& pair : entries) {
254 fs::path dir = pair.first.parent_path();
255 staged_by_directory[dir].push_back(pair.second);
256 }
257
258 std::map<fs::path, std::string> tree_hashes;
259
260 std::vector<fs::path> paths_to_process;
261 for (const auto& pair : staged_by_directory) {
262 paths_to_process.push_back(pair.first);
263 }
264
265 // Сортируем по длине пути (от самых длинных к самым коротким)
266 std::sort(paths_to_process.begin(), paths_to_process.end(), [](const fs::path& a, const fs::path& b) {
267 return a.generic_string().length() > b.generic_string().length();
268 });
269
270 for (const fs::path& current_dir : paths_to_process) {
271 std::vector<TreeEntry> current_tree_entries;
272
273 // Добавляем файлы из текущей директории
274 for (const IndexEntry& entry : staged_by_directory[current_dir]) {
275 if (entry.file_path.parent_path() == current_dir) {
276 TreeEntry te;
277 te.mode = "100644";
278 te.name = entry.file_path.filename().string();
279 te.hash_id = entry.blob_hash;
280 current_tree_entries.push_back(te);
281 }
282 }
283
284 // Добавляем поддиректории
285 for (auto it = tree_hashes.begin(); it != tree_hashes.end(); ) {
286 const fs::path& sub_tree_path = it->first;
287 const std::string& sub_tree_hash = it->second;
288
289 if (sub_tree_path.parent_path() == current_dir) {
290 TreeEntry te;
291 te.mode = "040000";
292 te.name = sub_tree_path.filename().string();
293 te.hash_id = sub_tree_hash;
294 current_tree_entries.push_back(te);
295
296 it = tree_hashes.erase(it);
297 } else {
298 ++it;
299 }
300 }
301
302 // Сортируем записи
303 std::sort(current_tree_entries.begin(), current_tree_entries.end());
304
305 Tree current_tree(current_tree_entries);
306 std::string tree_hash = current_tree.getHashId();
307 storage_.saveObject(current_tree);
308 tree_hashes[current_dir] = tree_hash;
309 }
310
311 // Ищем корневую директорию
312 auto root_it = tree_hashes.find(fs::path{});
313 if (root_it == tree_hashes.end()) {
314 throw std::runtime_error("Root tree object was not created successfully.");
315 }
316
317 return root_it->second;
318}
